% ------------- Title Page ------------------------------------------------
\usepackage{titling}

% Push the title block down
\setlength{\droptitle}{0.2\textheight}

\makeatletter
\renewcommand*{\maketitle}{%
  \newgeometry{top=50mm,bottom=40mm,left=35mm,right=35mm}

  \begin{titlepage}
    \begingroup
      \centering
      \vspace*{\droptitle}

      % --- Title ---------------------------------------------------------
      {\fontsize{44pt}{50pt}\selectfont\bfseries \thetitle\par}

      \vspace{0.5em}

      % --- Subtitle from Pandoc metadata ---------------------------------
      \makeatletter
      \ifdef{\@subtitle}{%
        {\fontsize{24pt}{28pt}\selectfont \@subtitle\par}
      }{}
      \makeatother

      \vspace{1em}

      % --- Year (comes in via Pandoc's \date) ----------------------------
      {\fontsize{32pt}{38pt}\selectfont \thedate\par}

      \vspace{2em}

      % --- Author --------------------------------------------------------
      {\fontsize{28pt}{32pt}\selectfont \theauthor\par}

      \vfill

      % --- Static tagline ------------------------------------------------
      {\fontsize{21pt}{24pt}\selectfont
       \textemdash\ Manuscript curation reference\ \textemdash\par}

      \vspace{1.5em}

      \thispagestyle{empty}
    \endgroup
  \end{titlepage}%
  \restoregeometry
}
\makeatother
